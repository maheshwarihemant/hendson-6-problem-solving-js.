Q.1 = Find the Smaller Angle

        var Minimal_Angle = (h, m) => 
{
  h=h%12;
  let hour = (h*360)/12 + (m*360)/(12*60);
  let min = (m*360)/60;
  let angle = Math.abs(hour-min);
  
  if(angle>180)
  return 360-angle;
  else
  return angle;
       
};

Q.3 = Perfect Number Check.

        var Perfect_Check = (N) => 
{
   let temp = 0;
   for( i=1;i<=N/2;i++)
     {
         if(N%i === 0)
          {
            temp += i;
          }
     }
   
     if(temp === N && temp !== 0)
        {
       return ("YES.");
        } 
     else
        {
       return ("NO");
        }   
        
};

Q.4 = Reverse a Number.

        var Reverse_Number = (N) => 
{
  let rev_num = 0;
  while(N > 0)
  {
    rev_num = rev_num *10 + N %10;
    N=Math.floor(N / 10);
  }
  return rev_num;
  

};
